#create a function for float-to-string transformation
def time_transform(df, time_var):
    #change data type from float to string and then zero-pad string.
    time_str = df[time_var].astype(str).replace('\.0', '', regex = True).str.zfill(4) 
    return time_str

def reformat_large_tick_values(tick_val, pos):
    if tick_val >= 1000:
        val = round(tick_val/1000, 0)
        new_tick_format = '{:,.0f}K'.format(val)
        # make new_tick_format into a string value
        tick_format = str(new_tick_format)
    else: 
        tick_format = str(tick_val)
    return tick_format
